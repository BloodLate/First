#include "DHT.h"

#define DHTPIN 13
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(115200);
  delay(1000);
  Serial.println("DHT11 start...");
  dht.begin();
}

void loop() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();

  if (isnan(h) || isnan(t)) {
    Serial.println("Failed to read from DHT11!");
  } else {
    Serial.print("温度: ");
    Serial.print(t);
    Serial.println(" C");

    Serial.print("湿度: ");
    Serial.print(h);
    Serial.println(" %");
    Serial.println("--------------------");
  }

  delay(2000);
}
